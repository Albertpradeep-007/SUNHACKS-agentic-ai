Agent 1 — Repo Miner (PyDriller)
Clones or pulls the repo, walks all commits in the last 90 days, extracts per-file churn and bug-fix signals, returns structured JSON. This is the data foundation for everything else.
backend/agents/miner.py
from pydriller import Repository
from datetime import datetime, timedelta
from collections import defaultdict
import re, git, os

BUG_PATTERN = re.compile(
    r'\b(fix|bug|patch|issue|defect|error|crash|hotfix|regression)\b',
    re.IGNORECASE
)
DAYS_BACK = 90

def clone_or_pull(repo_url: str, token: str, dest_dir: str) -> str:
    name = repo_url.split("/")[-1].replace(".git", "")
    path = os.path.join(dest_dir, name)
    auth_url = repo_url.replace("https://", f"https://{token}@")
    if os.path.exists(path):
        git.Repo(path).remotes.origin.pull()
    else:
        git.Repo.clone_from(auth_url, path)
    return path

def mine_repo(repo_path: str) -> list[dict]:
    since = datetime.now() - timedelta(days=DAYS_BACK)
    stats = defaultdict(lambda: {
        "module": "", "commit_count": 0, "bug_fix_count": 0,
        "churn": 0, "authors": set(), "last_modified": None
    })

    for commit in Repository(repo_path, since=since).traverse_commits():
        is_bugfix = bool(BUG_PATTERN.search(commit.msg))
        for f in commit.modified_files:
            if not f.new_path: continue
            key = f.new_path
            s = stats[key]
            s["module"]       = key
            s["commit_count"] += 1
            s["bug_fix_count"] += 1 if is_bugfix else 0
            s["churn"]        += (f.added_lines or 0) + (f.deleted_lines or 0)
            s["authors"].add(commit.author.email)
            s["last_modified"] = commit.author_date.isoformat()

    result = []
    for s in stats.values():
        s["authors"] = len(s["authors"])
        result.append(s)
    return sorted(result, key=lambda x: -x["churn"])
Agent 2 — Health Scorer (Radon)
Runs Radon on every Python file to get cyclomatic complexity and maintainability index. Combines with commit metrics using a weighted formula to produce a 0–100 health score.
backend/services/radon_runner.py
import subprocess, json, os

def get_radon_scores(repo_path: str) -> dict:
    """Returns {filepath: {"cc": float, "mi": float}}"""
    result = {}
    try:
        cc_out = subprocess.run(
            ["radon", "cc", repo_path, "-a", "-j"],
            capture_output=True, text=True, timeout=60
        )
        mi_out = subprocess.run(
            ["radon", "mi", repo_path, "-j"],
            capture_output=True, text=True, timeout=60
        )
        cc_data = json.loads(cc_out.stdout or "{}")
        mi_data = json.loads(mi_out.stdout or "{}")

        all_paths = set(cc_data.keys()) | set(mi_data.keys())
        for path in all_paths:
            rel = os.path.relpath(path, repo_path)
            blocks = cc_data.get(path, [])
            avg_cc = sum(b.get("complexity", 0) for b in blocks) / max(len(blocks), 1)
            mi = mi_data.get(path, {}).get("mi", 100)
            result[rel] = {"cc": avg_cc, "mi": mi}
    except Exception as e:
        print(f"Radon error: {e}")
    return result
backend/agents/scorer.py
from services.radon_runner import get_radon_scores

WEIGHTS = {"churn": 0.30, "bug_density": 0.30, "complexity": 0.25, "age": 0.15}

def compute_health_scores(raw_metrics: list, repo_path: str) -> list:
    radon = get_radon_scores(repo_path)
    scored = []
    max_churn = max((m["churn"] for m in raw_metrics), default=1)

    for m in raw_metrics:
        path = m["module"]
        r = radon.get(path, {"cc": 5, "mi": 50})

        churn_norm  = m["churn"] / max(max_churn, 1)
        bug_density = m["bug_fix_count"] / max(m["commit_count"], 1)
        cc_norm     = min(r["cc"] / 20.0, 1.0)   # 20+ CC is "very high"
        mi_norm     = 1 - (r["mi"] / 100.0)       # lower MI = worse
        complexity  = (cc_norm + mi_norm) / 2

        raw_risk = (
            WEIGHTS["churn"]       * churn_norm  +
            WEIGHTS["bug_density"] * bug_density +
            WEIGHTS["complexity"]  * complexity  +
            WEIGHTS["age"]        * mi_norm
        )
        score = round((1 - raw_risk) * 100, 1)   # 100 = perfect health
        tier  = "green" if score >= 70 else ("yellow" if score >= 40 else "red")

        scored.append({**m, "score": score, "tier": tier,
                        "complexity": round(complexity, 3),
                        "bug_density": round(bug_density, 3)})

    return sorted(scored, key=lambda x: x["score"])
Agent 3 — Risk Predictor (LLM + sklearn)
Uses linear regression on the health scores to project which modules cross the "critical" threshold in 90 days. Then sends results to GPT-4o for business-readable interpretation.
backend/agents/predictor.py
from sklearn.linear_model import LinearRegression
from langchain_openai import ChatOpenAI
from langchain.schema import HumanMessage, SystemMessage
import numpy as np, json

llm = ChatOpenAI(model="gpt-4o", temperature=0.2)

PREDICT_PROMPT = """
You are a senior engineering risk analyst.
Given module health scores and trends, identify:
1. Which modules will likely fail or cause incidents in 90 days
2. The probable business impact (outage, data loss, slow performance)
3. Estimated hours to remediate each

Return ONLY valid JSON: list of objects with keys:
module, risk_probability (0-1), predicted_failure_type, remediation_hours, urgency (high|medium|low)
"""

def predict_risks(health_scores: list) -> list:
    # Simple trend: simulate weekly score decay for high-risk modules
    predictions = []
    red_modules = [m for m in health_scores if m["tier"] != "green"][:10]

    for m in red_modules:
        score = m["score"]
        # Project: assume score degrades at (bug_density * 2) points/week
        decay_per_week = m["bug_density"] * 2 + m["complexity"] * 5
        projected_90d  = max(0, score - decay_per_week * 13)
        prob = min(1.0, (100 - projected_90d) / 100)
        predictions.append({**m, "projected_score_90d": projected_90d,
                             "risk_probability": round(prob, 2)})

    if not predictions: return []

    context = json.dumps(predictions, indent=2)
    response = llm.invoke([
        SystemMessage(content=PREDICT_PROMPT),
        HumanMessage(content=f"Modules:\n{context}")
    ])
    try:
        text = response.content.strip().lstrip("```json").rstrip("```")
        return json.loads(text)
    except:
        return predictions
Agent 4 — CEO Report Writer (LLM)
The money shot. Takes risk predictions and writes a non-technical business impact report. Then the PDF renderer converts it to a downloadable document.
backend/agents/reporter.py
from langchain_openai import ChatOpenAI
from langchain.schema import HumanMessage, SystemMessage
import json

llm = ChatOpenAI(model="gpt-4o", temperature=0.4)

CEO_PROMPT = """
You are a CTO writing a 1-page board-level report for a non-technical CEO.
Rules:
- NO technical jargon (no "cyclomatic complexity", no "churn rate")
- Translate risks into business consequences: lost revenue, outages, customer churn
- Give a cost-of-inaction figure in USD (assume $75/hr engineer, 8hr avg incident)
- Recommend a priority action list (max 4 items)
- Use confident, executive tone
- Format: markdown with sections: Executive Summary, Top Risks, Cost of Inaction, Recommended Actions

Inputs provided: repo name, risk predictions with remediation hours.
"""

ENG_HOURLY = 75

def generate_report(repo_name: str, predictions: list) -> str:
    total_hours = sum(p.get("remediation_hours", 16) for p in predictions)
    incident_cost = len([p for p in predictions if p.get("urgency") == "high"]) * 8 * ENG_HOURLY
    context = {
        "repo": repo_name,
        "total_remediation_hours": total_hours,
        "estimated_incident_cost_usd": incident_cost,
        "risks": predictions
    }
    response = llm.invoke([
        SystemMessage(content=CEO_PROMPT),
        HumanMessage(content=json.dumps(context, indent=2))
    ])
    return response.content

from jinja2 import Environment, FileSystemLoader
from weasyprint import HTML as WeasyprintHTML
import markdown, os

def render_pdf(report_md: str, output_path: str) -> str:
    html_body = markdown.markdown(report_md, extensions=["tables"])
    full_html = f"""
    <html><head><style>
      body {{ font-family: Arial, sans-serif; padding: 40px; color: #222; }}
      h1 {{ color: #1a1a2e; border-bottom: 2px solid #e63946; }}
      h2 {{ color: #457b9d; margin-top: 24px; }}
      table {{ border-collapse: collapse; width: 100%; margin: 16px 0; }}
      th, td {{ border: 1px solid #ccc; padding: 8px 12px; text-align: left; }}
      th {{ background: #1a1a2e; color: white; }}
    </style></head>
    <body>{html_body}</body></html>
    """
    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    WeasyprintHTML(string=full_html).write_pdf(output_path)
    return output_path

from fastapi import FastAPI, BackgroundTasks
from fastapi.middleware.cors import CORSMiddleware
from fastapi.responses import FileResponse
from pydantic import BaseModel
from database import Session, Repo, AnalysisRun, Base, engine
from agents.graph import run_pipeline
from datetime import datetime
import uvicorn, uuid, os

app = FastAPI(title="PEIP API")
app.add_middleware(CORSMiddleware, allow_origins=["*"],
                   allow_methods=["*"], allow_headers=["*"])
Base.metadata.create_all(engine)

class RepoCreate(BaseModel):
    url: str
    token: str

class AnalyzeRequest(BaseModel):
    repo_id: str

@app.post("/repos")
def add_repo(body: RepoCreate):
    with Session() as db:
        repo = Repo(id=str(uuid.uuid4()), url=body.url,
                    name=body.url.split("/")[-1], token=body.token)
        db.add(repo); db.commit()
        return {"id": repo.id, "name": repo.name}

@app.post("/analyze")
def start_analysis(body: AnalyzeRequest, bg: BackgroundTasks):
    run_id = str(uuid.uuid4())
    with Session() as db:
        run = AnalysisRun(id=run_id, repo_id=body.repo_id, status="pending")
        db.add(run); db.commit()
    bg.add_task(run_pipeline, run_id, body.repo_id)
    return {"run_id": run_id, "status": "started"}

@app.get("/results/{run_id}")
def get_results(run_id: str):
    with Session() as db:
        run = db.get(AnalysisRun, run_id)
        if not run: return {"error": "not found"}
        return {"status": run.status, "health_scores": run.health_scores,
                "predictions": run.predictions}

@app.get("/report/{run_id}/pdf")
def download_pdf(run_id: str):
    with Session() as db:
        run = db.get(AnalysisRun, run_id)
        if not run or not run.report_path:
            return {"error": "report not ready"}
        return FileResponse(run.report_path, media_type="application/pdf",
                            filename=f"peip_report_{run_id[:8]}.pdf")

if __name__ == "__main__":
    uvicorn.run("main:app", host="0.0.0.0", port=8000, reload=True)
backend/agents/graph.py — LangGraph orchestration
from langgraph.graph import StateGraph, END
from typing import TypedDict, Optional
from database import Session, AnalysisRun, Repo
from agents.miner import clone_or_pull, mine_repo
from agents.scorer import compute_health_scores
from agents.predictor import predict_risks
from agents.reporter import generate_report
from services.pdf_renderer import render_pdf
from datetime import datetime
import os

CLONE_DIR = os.getenv("CLONE_DIR", "/tmp/peip_repos")

class PipelineState(TypedDict):
    run_id: str; repo_id: str; repo_path: str
    raw_metrics: list; health_scores: list
    predictions: list; report_md: str

def node_mine(state: PipelineState) -> PipelineState:
    with Session() as db:
        repo = db.get(Repo, state["repo_id"])
    path = clone_or_pull(repo.url, repo.token, CLONE_DIR)
    metrics = mine_repo(path)
    state.update({"repo_path": path, "raw_metrics": metrics})
    return state

def node_score(state: PipelineState) -> PipelineState:
    scores = compute_health_scores(state["raw_metrics"], state["repo_path"])
    state["health_scores"] = scores; return state

def node_predict(state: PipelineState) -> PipelineState:
    state["predictions"] = predict_risks(state["health_scores"])
    return state

def node_report(state: PipelineState) -> PipelineState:
    with Session() as db:
        repo = db.get(Repo, state["repo_id"])
    md = generate_report(repo.name, state["predictions"])
    state["report_md"] = md; return state

graph = StateGraph(PipelineState)
graph.add_node("mine",    node_mine)
graph.add_node("score",   node_score)
graph.add_node("predict", node_predict)
graph.add_node("report",  node_report)
graph.set_entry_point("mine")
graph.add_edge("mine", "score")
graph.add_edge("score", "predict")
graph.add_edge("predict", "report")
graph.add_edge("report", END)
pipeline = graph.compile()

def run_pipeline(run_id: str, repo_id: str):
    with Session() as db:
        run = db.get(AnalysisRun, run_id)
        run.status = "running"; db.commit()
    try:
        result = pipeline.invoke({"run_id": run_id, "repo_id": repo_id})
        pdf_path = render_pdf(result["report_md"],
                              f"./reports/{run_id}.pdf")
        with Session() as db:
            run = db.get(AnalysisRun, run_id)
            run.raw_metrics = result["raw_metrics"]
            run.health_scores = result["health_scores"]
            run.predictions = result["predictions"]
            run.report_md   = result["report_md"]
            run.report_path = pdf_path
            run.status = "done"; run.finished = datetime.utcnow()
            db.commit()
    except Exception as e:
        with Session() as db:
            run = db.get(AnalysisRun, run_id)
            run.status = "error"; db.commit()
        raise

frontend/src/pages/Dashboard.jsx (core logic)
import { useState, useEffect } from 'react'
import { Treemap, ResponsiveContainer, Tooltip } from 'recharts'
import { getResults, reportUrl } from '../api'

const TIER_COLOR = { green: '#639922', yellow: '#BA7517', red: '#E24B4A' }

export default function Dashboard({ runId }) {
  const [data, setData] = useState(null)
  const [status, setStatus] = useState('loading')

  useEffect(() => {
    const poll = setInterval(async () => {
      const res = await getResults(runId)
      setStatus(res.data.status)
      if (res.data.status === 'done') {
        setData(res.data); clearInterval(poll)
      }
    }, 3000)
    return () => clearInterval(poll)
  }, [runId])

  if (status !== 'done') return <div>Analyzing... ({status})</div>

  const treemapData = data.health_scores.map(m => ({
    name: m.module.split('/').pop(),
    size: m.churn + 1,
    fill: TIER_COLOR[m.tier]
  }))

  return (
    <div>
      <ResponsiveContainer width="100%" height={400}>
        <Treemap data={treemapData} dataKey="size"
          content={({ x,y,w,h,name,fill }) =>
            <rect x={x} y={y} width={w} height={h} fill={fill} stroke="#fff"/>
          }
        />
      </ResponsiveContainer>
      <a href={reportUrl(runId)} target="_blank">
        <button>Download CEO Report (PDF)</button>
      </a>
    </div>
  )
}
